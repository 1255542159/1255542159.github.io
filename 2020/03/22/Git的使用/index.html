Java使用jdbc连接数据库

发表于2020-03-22|更新于2020-12-12
|阅读量:
public class MysqlDemo {
/*
java操作jdbc
 */

public static void main(String[] args) {
    try {
        //1.注册驱动(引入驱动包)
        //链接:[https://pan.baidu.com/s/1AIN0ESKkxGo8qV46wVegLw](https://pan.baidu.com/s/1AIN0ESKkxGo8qV46wVegLw)提取码：hd3i 

        Class.forName("com.mysql.jdbc.Driver");
        //2.获取连接
        Connection connection = DriverManager.getConnection("jdbc:mysql://localhost:3306/数据库名","数据库用户名","数据库密码");
        //3.获取statement对象
        Statement statement = connection.createStatement();
        //4.执行sql语句//得到结果集
        String sql = "select id,name,age,sex from student_1";
        ResultSet resultSet = statement.executeQuery(sql);
        //5.遍历结果集，取出数据
        while (resultSet.next()){  
            //根据列标签来获取值
            int id = resultSet.getInt("id");
            String name = resultSet.getString("name");
            int age = resultSet.getInt("age");
            String sex = resultSet.getString("sex");
            System.out.println(id+name+age+sex);
        }
        //6.先开的后关
        resultSet.close();
        statement.close();
        connection.close();
    } catch (ClassNotFoundException | SQLException e) {
        e.printStackTrace();
    }
}
}
